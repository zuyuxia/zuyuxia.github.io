    void swap(int &a, int &b){
    int temp = b;
    b = a;
    a = temp;
}

冒泡排序

  • 特点：

    • 时间复杂度：最坏O(N^2)，最好O(N)，平均O(N^2)
    • 空间复杂度：O(1)
    • 稳定性：稳定

  • 基本思路

    • 交换排序，比较两个相邻的元素，如果反序则交换，直到没有反序的元素为止

    • 优化方法：设置一个flag，标记元素是否有序

  • 代码： 

    void BublbleSort(vector<int> &nums){
    if(nums.empty())
        return;
    bool flag = true;
    for(int i = 0; flag && i < nums.size()-1; ++i){
        flag = false;
        for(int j = nums.size()-1; j > i; --j){
            if(nums[j-1] > nums[j]){
                swap(nums[j-1], nums[j]);
                flag = true;
            }
        }
    }
}

快速排序

  • 特点：

    • 时间复杂度：最坏O(N^2)，最好O(NlogN)，平均O(NlogN)
    • 空间复杂度：O(logN)
    • 稳定性：不稳定

  • 基本思路

    • 通过一趟排序将待排序的元素分为两组，一组的关键字比另一组的关键字都小，然后再分别对这两组继续进行排序，最后全都有序

    • 基准数的选择是重点：固定位置选取，随机选取基准数，三数取中

  • 代码：

    void QuickSort(vector<int> &nums, int begin, int end){
    if(nums.empty() || begin >= end)
        return;
    int pivot = Partition(nums, begin, end);
    QuickSort(nums, begin, pivot - 1);
    QuickSort(nums, pivot + 1, end);
}

int Partiton(vector<int> &nums, int begin, int end){
    int i = begin + 1, j = end;
    int mark = nums[begin];
    while(i < j){
        while(i < j && nums[j] > mark)
            --j;
        while(i < j && nums[i] < mark)
            ++i;
        if(i < j)
            swap(nums[i],nums[j]);
    }
    swap(nums[i],nums[begin]);
    return i;
}

归并排序

  • 特点：

    • 时间复杂度：最坏O(NlogN)，最好O(NlogN)，平均O(NlogN)
    • 空间复杂度：O(N)
    • 稳定性：稳定

  • 基本思路

    • 将N个元素看成是N个子序列，每两个子序列进行合并，得到N/2个子序列，再两两进行合并，重复直到序列长度为N

    • 分治法的典型应用

  • 代码：

    void MergeSort(vector<int> &nums){
    if(nums.empty() || nums.size() == 1)
        return;
    vector<int> temp = nums;
    MSort(nums, 0, nums.size()-1, temp);
}

void Merge(vector<int> &nums, int begin, int middle, int end, vector<int> &temp){
    int i = begin, j = middle + 1, k = begin;
    while(i <= middle && j <= end){
        if(nums[i] < nums[j]){
            temp[k++] = nums[i++];
        }
        else{
            temp[k++] = nums[j++];
        }
    }
    while(i <= middle){
        temp[k++] = nums[i++];
    }
    while(j <= end){
        temp[k++] = nums[j++];
    }
    for(int l = begin; l <= end; ++l)
        nums[l] = temp[l];
}

堆排序

  • 特点：

    • 时间复杂度：最坏O(NlogN)，最好O(NlogN)，平均O(NlogN)
    • 空间复杂度：O(1)
    • 稳定性：不稳定

  • 基本思路

    • 利用堆进行排序，先将待排序的元素构造成一个堆（以大顶堆为例），然后将堆顶元素与最后一个元素交换，将剩下的N-1个元素重新构造成一个堆，重复操作

    • 堆是完全二叉树，每个结点的值都大于或等于（小于或等于）其左右孩子结点的值

  • 代码：

    void HeapSort(vector<int> &nums){
    if(nums.empty() || nums.size() == 1)
        return;

    //将待排序数组构成成一个堆
    for(int i = nums.size()/2 - 1;i >= 0; --i)
        HeapAddjust(nums,i,nums.size());

    for(int i = 0; i < nums.size() - 1;++i){
        swap(nums[i],nums[nums.size()-1-i]);
        HeapAddjust(nums,0,nums.size()-1-i);
    }

}

void HeapAddjust(vector<int> &nums, int parentIndex, int length){
    int temp = nums[parentIndex];
    int leftChild = parentIndex*2 + 1;
    for(;leftChild < length - 1;leftChild = leftChild*2 + 1){
        if(leftChild + 1 < length - 1 && nums[leftChild] < nums[leftChild+1])
            ++leftChild;

        if(temp == nums[leftChild])
            break;

        nums[parentIndex] = nums[leftChild];
        parentIndex = leftChild;
    }
    nums[parentIndex] = temp;
}

桶排序

  • 特点：

    • 时间复杂度：最坏O(N^2)，最好O(N+k)，平均O(N+k)
    • 空间复杂度：O(KN)
    • 稳定性：稳定

  • 基本思路

    • 假定元素处在一个范围内，划分多个区间，将这些元素分别放到这些区间，然后再分别对这些区间里面的元素进行排序

    • 空间复杂度很高，同时还要合理的设置桶

  • 代码：（以下例简单介绍桶排序代码：将元素位于[0,99]之间的的数组进行排序，其中无重复元素）

    void BucketSort(vector<int> &nums){
    if(nums.empty() || nums.size() == 1)
        return;

    //设置10个桶
    vector<vector<int>> bucket;
    for(int i = 0;i < 10;++i)
        bucket.push_back({});

    //将数组中的元素分别置入相应的桶中
    for(int i = 0;i < nums.size();++i){
        bucket[nums[i]/10].push_back(nums[i]);
    }

    //对各个桶中的元素进行排序
    for(int i = 0;i < 10;++i)
        sort(bucket[i].begin(),bucket[i].end());

    //将桶中有序的元素写回数组之中
    int start = 0;
    for(int i = 0;i < 10;++i){
        for(int j = 0;j < bucket[i].size();++j)
            nums[start++] = bucket[i][j];
    }

}

  • 上面的例子只是用来简单的说明桶排序的思想，实际上在桶内的排序往往有各种各样的选择，应该依据实际情况进行选择

插入排序

  • 特点：

    • 时间复杂度：最坏O(N^2)，最好O(N)，平均O(N^2)
    • 空间复杂度：O(1)
    • 稳定性：稳定

  • 基本思路

    • 依次将一个元素插入到已经排序好的有序序列中

    • 类比扑克牌

  • 代码：

    void InsertSort(vector<int> &nums){
    if(nums.empty() || nums.size() == 1)
        return;

    int mark, j;
    for(int i = 1;i < nums.size();++i){
        if(nums[i] < nums[i-1]){
            mark = nums[i];
            for(j = i; j > 0 && nums[j] > mark;--j)
                nums[j] = nums[j-1];
            nums[j] = mark;
        }
    }
}

选择排序

  • 特点：

    • 时间复杂度：最坏O(N^2)，最好O(N^2)，平均O(N^2)
    • 空间复杂度：O(1)
    • 稳定性：不稳定

  • 基本思路

    • 每一趟比较中，选取关键字最小的元素

  • 代码：

    void SelectSort(vector<int> &nums){
    if(nums.empty() || nums.size() == 1)
        return;

    for(int i = 0;i < nums.size()-1;++i){
        int minIndex = i;
        for(int j = i+1;j < nums.size();++j){
            if(nums[j] < nums[minIndex])
                minIndex = j;
        }
        if(minIndex != i)
            swap(nums[i],nums[minIndex]);
    }
}

shell排序

  • 特点：

    • 时间复杂度：最坏O(NLogN)，最好O(N)，平均O(N^1.3)
    • 空间复杂度：O(1)
    • 稳定性：不稳定

  • 基本思路

    • 将相距某个增量的元素组成一个子序列，对子序列进行直接插入排序，不断减小增量，重复前述操作，最后完成排序

    • 跳跃分割，如何选择增量还存在难题
